\begin{longtable}{|p{4cm}|X|}
	\caption{\label{tab:securitylevels}Mögliche Unterteilung des Härtens in verschieden Kategorien}\\
	\hline \textbf{Kategorie} & \textbf{Beschreibung} \\
	\hline
	\hline  Physisch & Der physische Zugang zum System wird verhindert bzw. geregelt (z.B. wird der Computer in einen Schrank eingeschlossen). \\
	\hline Lokal & Die Rechte der lokalen Benutzer (user mit einem account) werden gesetzt.\\
	\hline  Dateien und Dateisysteme& Es wird ein geeignetes Dateisystem gewählt und Dateiattribute werden gesetzt. \\
	\hline  Passwörter und Verschlüsselung& Passwörter werden überprüft und es wird wenn immer möglich Verschlüsselung eingesetzt. \\
	\hline Betriebssystem-Kern & Kernel Optionen werden in Hinblick auf die Sicherheit entsprechend gesetzt bzw. verändert.\\
	\hline  Netzwerk & Der Zugriff über ein Netzwerk wird so sicher wie möglich gemacht. \\
	\hline
\end{longtable}
